add_action('wp_footer', 'my_custom_script', 9999);

function my_custom_script() {
    $check_url = home_url('/checkout');


    if (is_checkout()) {
        ?>
        <script>
            document.addEventListener("DOMContentLoaded", function() {


                // Function to check and update th elements
                function checkAndUpdateThElements() {
                    const existingRow = document.querySelector('tr.cart-subtotal-after-discount');
                    if (!existingRow) {
                        const rowElement = document.querySelector('tr');

                        const totalAmountElement = document.querySelector('.cart-subtotal td .woocommerce-Price-amount bdi');
                        // // console.log(totalAmountElement);
                        const totalAmount = parseFloat(totalAmountElement.textContent.replace(/[^0-9.-]+/g, ''));
                        //
                        console.log("Total",totalAmount);
                        const existRow = document.querySelector('tr.cart-discount span.woocommerce-Price-amount.amount');
                        if(existRow){
                        // // Get the discount amount
                        const discountAmountElement = document.querySelector('tr.cart-discount span.woocommerce-Price-amount.amount').textContent;
                        // cosole.log(discountAmountElement)
                        const discountAmount = parseFloat(discountAmountElement.replace(/[^0-9.-]+/g, ''));
                        console.log("Total",discountAmount);
                        //
                        // // Calculate subtotal after discount
                        const subtotalAfterDiscount = totalAmount - discountAmount;
                        console.log(subtotalAfterDiscount)

                        let container = document.getElementsByClassName("shop_table.woocommerce-checkout-review-order-table");

                        const newRow = document.createElement('tr');
                        newRow.classList.add('cart-subtotal-after-discount');
                        //
                        const thElement = document.createElement('th');
                        thElement.textContent = 'Subtotal (after discount applied)';

                        const tdElement = document.createElement('td');
                        //
                        const spanElement = document.createElement('span');
                        spanElement.classList.add('calculated-subtotal');
                        spanElement.textContent = '$' + subtotalAfterDiscount.toFixed(2);
                        //
                        console.log(spanElement);
                        tdElement.appendChild(spanElement);
                        console.log(tdElement);
                        newRow.appendChild(thElement);
                        newRow.appendChild(tdElement);
                        console.log(newRow);
                        //
                        const discountRows = document.querySelectorAll('.cart-discount');
                        //
                        const lastDiscountRow = discountRows[discountRows.length - 1];
                        //
                        lastDiscountRow.parentNode.insertBefore(newRow, lastDiscountRow.nextSibling);

                        console.log(lastDiscountRow);
                    }
                    }


                    const thElements = document.querySelectorAll("th");
                    // console.log(thElements);

                    thElements.forEach(th => {

                        if (th.textContent.includes("Coupon")) {
                            const couponValue = th.textContent.split(":")[1].trim();

                            th.textContent = "Coupon/Promotion Applied: " + couponValue;
                        }
                    });
                }

                checkAndUpdateThElements();

                setInterval(checkAndUpdateThElements, 100);
            });
        </script>
        <?php
    }
}
